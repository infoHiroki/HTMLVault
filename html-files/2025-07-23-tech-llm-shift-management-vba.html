2. 今回作るもの：シンプルなシフト管理ツール

🎯 完成イメージ

こんなシフト表がボタン一つで自動作成されます：

  • スタッフ5名の1ヶ月分シフト
  • 休日を自動でランダム配置（各自月8日）
  • 出勤は緑、休みは赤で色分け
  • 手作業なら30分かかる作業が3秒で完了

3. 実際のコード（コピペで使える！）

💡 コードの意味が分からなくても大丈夫！
まずはコピペして動かしてみましょう。動いたら少しずつ理解していけばOKです。

📋 完全版コード

以下のコードをそのままコピーして使えます：

' シフト管理システム

Option Explicit

Sub シフト管理開始()
    Dim 選択 As String
    選択 = InputBox("何をしますか？" & vbNewLine & _
                   "1: シート作成" & vbNewLine & _
                   "2: シフト作成" & vbNewLine & _
                   "番号を入力:")
    
    If 選択 = "1" Then
        シート作成
    ElseIf 選択 = "2" Then
        シフト作成
    Else
        MsgBox "終了します"
    End If
End Sub

Sub シート作成()
    On Error GoTo エラー処理
    
    ' 既存シート削除
    Dim ws As Worksheet
    Application.DisplayAlerts = False
    For Each ws In Worksheets
        If ws.Name = "スタッフ" Or ws.Name = "シフト" Then
            ws.Delete
        End If
    Next ws
    Application.DisplayAlerts = True
    
    ' スタッフシート作成
    Dim スタッフシート As Worksheet
    Set スタッフシート = Sheets.Add
    スタッフシート.Name = "スタッフ"
    
    ' ヘッダー設定
    スタッフシート.Range("A1:C1").Value = Array("番号", "名前", "希望休日数")
    
    ' サンプルデータ
    Dim i As Integer
    For i = 2 To 6
        スタッフシート.Cells(i, 1).Value = i - 1
        スタッフシート.Cells(i, 2).Value = "スタッフ" & (i - 1)
        スタッフシート.Cells(i, 3).Value = 8  ' 月8日休み希望
    Next i
    
    ' シフトシート作成
    Dim シフトシート As Worksheet
    Set シフトシート = Sheets.Add
    シフトシート.Name = "シフト"
    
    MsgBox "シート作成完了！" & vbNewLine & "スタッフシートに名前を入力してください"
    Exit Sub
    
エラー処理:
    MsgBox "エラー: " & Err.Description
End Sub

Sub シフト作成()
    On Error GoTo エラー処理
    
    ' シート確認
    Dim スタッフシート As Worksheet
    Dim シフトシート As Worksheet
    
    On Error Resume Next
    Set スタッフシート = Sheets("スタッフ")
    Set シフトシート = Sheets("シフト")
    On Error GoTo エラー処理
    
    If スタッフシート Is Nothing Or シフトシート Is Nothing Then
        MsgBox "先にシート作成を実行してください"
        Exit Sub
    End If
    
    ' シフト表クリア
    シフトシート.Cells.Clear
    
    ' 月の設定
    Dim 開始日 As Date
    開始日 = DateSerial(Year(Date), Month(Date), 1)
    
    ' ヘッダー作成
    シフトシート.Cells(1, 1).Value = "名前"
    
    Dim 日付列 As Integer
    Dim 日付 As Date
    
    For 日付列 = 2 To 32
        日付 = 開始日 + 日付列 - 2
        If Month(日付) <> Month(開始日) Then Exit For
        シフトシート.Cells(1, 日付列).Value = Day(日付)
    Next 日付列
    
    ' スタッフ名をコピー
    Dim スタッフ行 As Integer
    Dim 最終行 As Integer
    最終行 = スタッフシート.Cells(Rows.Count, 2).End(xlUp).Row
    
    For スタッフ行 = 2 To 最終行
        シフトシート.Cells(スタッフ行, 1).Value = スタッフシート.Cells(スタッフ行, 2).Value
    Next スタッフ行
    
    ' 簡単なシフト割り当て（ランダム）
    Dim 休日数 As Integer
    Dim 割当日 As Integer
    
    For スタッフ行 = 2 To 最終行
        休日数 = スタッフシート.Cells(スタッフ行, 3).Value
        
        ' ランダムに休日を割り当て
        Dim 休日カウント As Integer
        休日カウント = 0
        
        Do While 休日カウント < 休日数
            割当日 = Int((日付列 - 2) * Rnd + 2)
            
            If シフトシート.Cells(スタッフ行, 割当日).Value = "" Then
                シフトシート.Cells(スタッフ行, 割当日).Value = "休"
                シフトシート.Cells(スタッフ行, 割当日).Interior.Color = RGB(255, 200, 200)
                休日カウント = 休日カウント + 1
            End If
        Loop
        
        ' 残りは出勤
        For 日付列 = 2 To 31
            If シフトシート.Cells(1, 日付列).Value = "" Then Exit For
            If シフトシート.Cells(スタッフ行, 日付列).Value = "" Then
                シフトシート.Cells(スタッフ行, 日付列).Value = "出"
                シフトシート.Cells(スタッフ行, 日付列).Interior.Color = RGB(200, 255, 200)
            End If
        Next 日付列
    Next スタッフ行
    
    MsgBox "シフト作成完了！"
    Exit Sub
    
エラー処理:
    MsgBox "エラー: " & Err.Description
End Sub

4. 使い方（たった4ステップ）

🚀 セットアップ手順

⚠️ 開発タブが表示されていない場合
1. ファイル → オプション → リボンのユーザー設定
2. 右側の「開発」にチェックを入れてOK
  1. Excelを開く
  2. Alt + F11 を押す（VBAエディタが開きます）
  3. 挿入 → 標準モジュール をクリック
  4. 上のコードをコピペして貼り付け

📝 実行方法

  1. F5キーを押すか、実行ボタンをクリック
  2. 「シフト管理開始」を選んで実行
  3. 最初に「1」を入力してシートを作成
  4. 次に「2」を入力してシフトを自動生成
✅ これだけで完成！
「スタッフ」シートで名前を変更したり、人数を増やしたりできます。
